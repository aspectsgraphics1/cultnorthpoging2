<!DOCTYPE html>
<html>
<head>
  <style>
    /* Stijl voor de videocontainer */
    #video-container {
      display: none;
    }
  </style>
</head>
<body>
  <!-- GIF -->
  <img id="gif" src="path/to/your/gif.gif" alt="GIF">

  <!-- Video -->
  <div id="video-container">
    <video id="video" width="640" height="360" controls>
      <source src="path/to/your/video.mp4" type="video/mp4">
    </video>
  </div>

  <script>
    // Wacht tot het document volledig is geladen
    document.addEventListener('DOMContentLoaded', function() {
      var gif = document.getElementById('gif');
      var videoContainer = document.getElementById('video-container');
      var video = document.getElementById('video');

      // Klikgebeurtenis op de GIF
      gif.addEventListener('click', function() {
        // Verberg de GIF en toon de video
        gif.style.display = 'none';
        videoContainer.style.display = 'block';

        // Speel de video af
        video.play();
      });

      // Gebeurtenis wanneer de video is beÃ«indigd
      video.addEventListener('ended', function() {
        // Toon de GIF en verberg de video
        gif.style.display = 'block';
        videoContainer.style.display = 'none';

        // Start de GIF opnieuw
        gif.src = gif.src;
      });
    });
  </script>
</body>
</html>
